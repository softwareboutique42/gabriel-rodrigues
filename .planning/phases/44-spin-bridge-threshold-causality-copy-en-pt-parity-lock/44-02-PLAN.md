**CRITICAL anti-patterns to avoid (from RESEARCH.md):**
- Do NOT inject both spins in a single `page.evaluate` — inject spin 1, await attribute assertion, then inject spin 2 (prevents race condition per STATE.md blocker)
- Set BOTH `ccz-tutorial-completed` AND `ccz-near-miss-completed` in localStorage before reload (the unlock chain requires both — per RESEARCH.md Pitfall 4)
- Use exact `version: 1` format for bridge events: `{ type: 'ccz:spin-settled', version: 1, payload: { spinIndex: N } }`

**File structure:**

```typescript
import { test, expect } from '@playwright/test';

test.describe('Casinocraftz Lesson 3 — spin-bridge gate (EDU-72)', () => {

  test('sensory-conditioning-observe does not advance before 2 spins and advances on the second — EN', async ({ page }) => {
    // 1. Navigate to /en/casinocraftz/
    await page.goto('/en/casinocraftz/');

    // 2. Set localStorage for returning user with Lesson 2 complete (D-06)
    await page.evaluate(() => {
      localStorage.setItem('ccz-tutorial-completed', '1');
      localStorage.setItem('ccz-near-miss-completed', '1');
    });
    await page.reload();

    // 3. Locate shell root
    const root = page.locator('[data-casinocraftz-shell-root]');

    // 4. Open Lesson 3 (sensory-conditioning) — click the lesson action button
    await page.locator('[data-casinocraftz-lesson="sensory-conditioning"] [data-casinocraftz-lesson-action="sensory-conditioning"]').click();

    // 5. Advance through intro step to reach observe step
    await page.locator('[data-casinocraftz-tutorial-next]').click();
    await expect(root).toHaveAttribute('data-casinocraftz-tutorial-step', 'sensory-conditioning-observe');

    // 6. Inject spin 1 — step must NOT advance
    await page.evaluate(() => {
      window.postMessage({ type: 'ccz:spin-settled', version: 1, payload: { spinIndex: 0 } }, '*');
    });
    // Wait and assert step is STILL sensory-conditioning-observe (the "does not advance" half)
    await expect(root).toHaveAttribute('data-casinocraftz-tutorial-step', 'sensory-conditioning-observe');

    // 7. Inject spin 2 — step MUST advance to sensory-conditioning-reveal
    await page.evaluate(() => {
      window.postMessage({ type: 'ccz:spin-settled', version: 1, payload: { spinIndex: 1 } }, '*');
    });
    await expect(root).toHaveAttribute('data-casinocraftz-tutorial-step', 'sensory-conditioning-reveal');
  });

  test('causality disclosure renders after spin-triggered transition — EN and PT', async ({ page }) => {
    for (const lang of ['en', 'pt'] as const) {
      await page.goto(`/${lang}/casinocraftz/`);

      // Set localStorage for returning user
      await page.evaluate(() => {
        localStorage.setItem('ccz-tutorial-completed', '1');
        localStorage.setItem('ccz-near-miss-completed', '1');
      });
      await page.reload();

      const root = page.locator('[data-casinocraftz-shell-root]');

      // Open Lesson 3
      await page.locator('[data-casinocraftz-lesson="sensory-conditioning"] [data-casinocraftz-lesson-action="sensory-conditioning"]').click();

      // Advance to observe step
      await page.locator('[data-casinocraftz-tutorial-next]').click();
      await expect(root).toHaveAttribute('data-casinocraftz-tutorial-step', 'sensory-conditioning-observe');

      // Inject 2 spins to trigger transition to reveal
      await page.evaluate(() => {
        window.postMessage({ type: 'ccz:spin-settled', version: 1, payload: { spinIndex: 0 } }, '*');
      });
      await expect(root).toHaveAttribute('data-casinocraftz-tutorial-step', 'sensory-conditioning-observe');

      await page.evaluate(() => {
        window.postMessage({ type: 'ccz:spin-settled', version: 1, payload: { spinIndex: 1 } }, '*');
      });
      await expect(root).toHaveAttribute('data-casinocraftz-tutorial-step', 'sensory-conditioning-reveal');

      // Causality disclosure must be visible after spin-triggered transition
      // (lastTransitionTrigger === 'spin' gates this — per main.ts recap logic)
      await expect(page.locator('[data-casinocraftz-recap="true"]')).toBeVisible();
    }
  });

});
```
