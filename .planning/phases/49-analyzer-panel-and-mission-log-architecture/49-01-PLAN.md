```css
/* === CASINOCRAFTZ LOBBY LAYOUT (Phase 49) === */

/* Two-column desktop grid: [260px sidebar] [1fr main] */
.ccz-lobby-grid {
  display: grid;
  grid-template-columns: 260px 1fr;
  gap: 1.5rem;
  align-items: start;
}

/* Sticky sidebar — follows scroll on desktop */
.ccz-analyzer-panel {
  position: sticky;
  top: 1.5rem;
}

/* Mobile: single column, no sticky */
@media (max-width: 639px) {
  .ccz-lobby-grid {
    grid-template-columns: 1fr;
  }
  .ccz-analyzer-panel {
    position: static;
  }
}

/* Desktop: always show analyzer drawer regardless of hidden attribute.
   The hidden attr is used for mobile collapse; CSS overrides it on desktop.
   This prevents flash-of-collapsed-content on desktop before JS runs. */
@media (min-width: 640px) {
  [data-ccz-analyzer-drawer][hidden] {
    display: flex !important;
    flex-direction: column;
    gap: 0.5rem;
  }
}
```
