```typescript
function mountMissionLog(root: HTMLElement): void {
  const toggle = root.querySelector<HTMLButtonElement>('[data-ccz-mission-log-toggle]');
  const content = root.querySelector<HTMLElement>('[data-ccz-mission-log-content]');
  if (!toggle || !content) return;

  const isOpen = sessionStorage.getItem('ccz-mission-log-open') === '1';
  if (isOpen) {
    content.removeAttribute('hidden');
    toggle.setAttribute('aria-expanded', 'true');
    toggle.textContent = toggle.dataset.labelClose ?? toggle.textContent;
  }

  toggle.addEventListener('click', () => {
    const willOpen = content.hasAttribute('hidden');
    if (willOpen) {
      content.removeAttribute('hidden');
      toggle.setAttribute('aria-expanded', 'true');
      toggle.textContent = toggle.dataset.labelClose ?? '';
      sessionStorage.setItem('ccz-mission-log-open', '1');
    } else {
      content.setAttribute('hidden', '');
      toggle.setAttribute('aria-expanded', 'false');
      toggle.textContent = toggle.dataset.labelOpen ?? '';
      sessionStorage.setItem('ccz-mission-log-open', '0');
    }
  });
}

function mountAnalyzerDrawer(root: HTMLElement): void {
  const toggle = root.querySelector<HTMLButtonElement>('[data-ccz-analyzer-toggle]');
  const drawer = root.querySelector<HTMLElement>('[data-ccz-analyzer-drawer]');
  if (!toggle || !drawer) return;

  // Mobile-only: desktop visibility is handled entirely by CSS (Plan 01 global.css rule)
  if (window.innerWidth >= 640) return;

  const isOpen = sessionStorage.getItem('ccz-analyzer-open') === '1';
  if (isOpen) {
    drawer.removeAttribute('hidden');
    toggle.setAttribute('aria-expanded', 'true');
    toggle.textContent = toggle.dataset.labelClose ?? toggle.textContent;
  }

  toggle.addEventListener('click', () => {
    const willOpen = drawer.hasAttribute('hidden');
    if (willOpen) {
      drawer.removeAttribute('hidden');
      toggle.setAttribute('aria-expanded', 'true');
      toggle.textContent = toggle.dataset.labelClose ?? '';
      sessionStorage.setItem('ccz-analyzer-open', '1');
    } else {
      drawer.setAttribute('hidden', '');
      toggle.setAttribute('aria-expanded', 'false');
      toggle.textContent = toggle.dataset.labelOpen ?? '';
      sessionStorage.setItem('ccz-analyzer-open', '0');
    }
  });
}
```

Then at the END of the existing `mountLobby()` function body (before the closing `}`), add:
